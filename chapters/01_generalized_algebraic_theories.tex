\section{Definition}

We assume we have a set $V$ of countably many distinct variables.
Let $W$ be an \textbf{alphabet}, and its elements be \textbf{symbols}.

\begin{definition}
    The set of \textbf{expressions} is defined inductively such that they are finite
    sequences of elements of $W \cup V \cup \{ ( \} \cup \{ ) \} \cup \{ , \}$
    by the clauses
    \begin{enumerate}
        \item if $x \in V$, then $x$ is an expression;
        \item if $L \in W$, then $L$ is an expression;
        \item if $L \in W$ and $e_1, \dots, e_n$ are expressions,
        then $L(e_1, \dots, e_n)$ is an expression.
    \end{enumerate}
\end{definition}

\begin{definition}
    A \textbf{premise} is defined to be a sequence of $V \times \text{set of expressions}$,
    the empty sequence is allowed and is called the \textbf{empty premise}.
    A premise $((x_1, \Delta_1), \dots, (x_n, \Delta_n))$ is written as
    $x_1 : \Delta_1, \dots, x_n : \Delta_n$.
\end{definition}

\begin{definition}
    A \textbf{conclusion} is one of the following:
    \begin{enumerate}
        \item \textbf{$T$-conclusion} consists a single expression $\Delta$ 
            and is written as "$\Delta$ is a type";
        \item \textbf{$\in$-conclusion} consists of pair of expressions $(t, \Delta)$
            and is written as "$t : \Delta$";
        \item \textbf{$T=$conclusion} consists of a pair of expressions $(\Delta_1, \Delta_2)$
            and is written as "$\Delta_1 = \Delta_2$";
        \item \textbf{$\in=$conclusion} consists of a triple of expressions $(t_1, t_2, \Delta)$
            and is written as "$t_1 = t_2 \in \Delta$".
    \end{enumerate}
\end{definition}

\begin{definition}
    A \textbf{rule} is determined by a premise $P$ and a conclusion $C$, and is written as
    \begin{prooftree}
        \AxiomC{P}
        \UnaryInfC{C}
    \end{prooftree}
\end{definition}

If $\Delta, t_1, \dots, t_n$ expressions and $x_1, \dots, x_n$, then the expression 
$\Delta[t_1 \vert x_1, \dots, t_n \vert x_n]$ is the result of simultaneous replacement 
of $x_1, \dots, x_n$ in $\Delta$ by $t_1, \dots, t_n$.

\begin{definition}
    A \textbf{pretheory} consists of
    \begin{enumerate}
        \item a set $S$, of sort symbols;
        \item a set $Z$, of operator symbols;
        \item for each sort symbol $A$, an associated rule of the alphabet $S \cup Z$, 
            called \textbf{introductory rule} for $A$, of the form
            \begin{prooftree}
                \AxiomC{$x_1 : \Delta_1, \dots, x_n : \Delta_n$}
                \UnaryInfC{$A(x_1, \dots, x_n)$ is a type}
            \end{prooftree}
            for some $n \geq 0$;
        \item for each operator symbol $F$, an associated rule of the alphabet $S \cup Z$,
            called \textbf{introductory rule} for $F$, of the form
            \begin{prooftree}
                \AxiomC{$x_1 : \Delta_1, \dots, x_n : \Delta_n$}
                \UnaryInfC{$F(x_1, \dots, x_n) \in \Delta$}
            \end{prooftree}
            for each $n \geq 0$;
        \item a set of axioms, each axiom is either a $T=$rule or a $\in$-rule of
            the alphabet $S \cup Z$.
    \end{enumerate}
\end{definition}

\begin{definition}
    If $U$ is a pretheory, then
    \begin{enumerate}
        \item a \textbf{context} is a premise $x_1 : \Delta_1, \dots, x_n : \Delta_n$ such that the rule
            \begin{equation*}
                \dfrac{x_1 : \Delta_1, \dots, x_{n-1} : \Delta_{n-1}}{\Delta_n \text{is a type}}
            \end{equation*}
            is a derived rule such that $x_n$ is distinct from all of $x_1, \dots, x_{n-1}$.
        \item TODO
    \end{enumerate}
\end{definition}

\begin{definition}
    The set of \textbf{derived rules} of $U$ is the set of rules derivable by 
    the following principles of derivation
    \begin{itemize}
        \item[LI1] From $\dfrac{P}{\Delta \text{ is a type}}$ derive $\dfrac{P}{\Delta = \Delta}$.
        \item[LI2] From $\dfrac{P}{t : \Delta}$ derive $\dfrac{P}{t = t : \Delta}$.
        \item[LI3] From $\dfrac{P}{\Delta_1 = \Delta_2}$ derive $\dfrac{P}{\Delta_2 = \Delta_1}$.
        \item[LI4] From $\dfrac{P}{t_1 = t_2 : \Delta}$ derive $\dfrac{P}{t_2 = t_1 : \Delta}$.
        \item[LI5] From $\dfrac{P}{\Delta_1 = \Delta_2}$ and $\dfrac{P}{\Delta_2 = \Delta_3}$ derive $\dfrac{P}{\Delta_1 = \Delta_3}$.
        \item[LI6] From $\dfrac{P}{t_1 = t_2 : \Delta}$ and $\dfrac{P}{t_2 = t_3 : \Delta}$ derive $\dfrac{P}{t_1 = t_3 : \Delta}$.
        \item[LI7] From $\dfrac{P}{t_1 = t_2 : \Delta_1}$ and $\dfrac{P}{\Delta_1 = \Delta_2}$ derive $\dfrac{P}{t_1 = t_2 : \Delta_2}$.
        \item[T1] From $\dfrac{P}{\Delta_1 = \Delta_2}$ and $\dfrac{P}{t : \Delta_1}$ derive $\dfrac{P}{t : \Delta_2}$.
        \item[CF1] For $n \geq 0$, $1 \leq i \leq n + 1$: \\
            From $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta_{n+1} \text{is a type}}$ derive 
            $\dfrac{x_1 : \Delta_1, \dots, x_{n+1} : \Delta_{n+1}}{x_i : \Delta_i}$. \\
            providing that $x_{n+1}$ is a variable distinct from all of $x_1, \dots, x_n$.
        \item[CF2(a)] For every sort symbol $A$ with well-formed introductory rule \\
            $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{A(x_1, \dots, x_n) \text{is a type}}$, for every context $P$, \\
            from $\dfrac{P}{t_1 : \Delta_1}, \dfrac{P}{t_2 : \Delta[t_1 \vert x_1]}, \dots, \dfrac{P}{t_n : \Delta_n[t_1 \vert x_1, \dots, t_{n-1} \vert x_{n-1}]}$ \\
            derive $\dfrac{P}{A(t_1, \dots, t_n) \text{is a type}}$.
        \item[CF2(b)] For every operator symbol $F$ with well-formed introductory rule \\
            $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{F(x_1, \dots, x_n) : \Delta}$, for every context $P$, \\
            from $\dfrac{P}{t_1 : \Delta_1}, \dfrac{P}{t_2 : \Delta_2[t_1 \vert x_1]}, \dots, \dfrac{P}{t_n : \Delta_n[t_1 \vert x_1, \dots, t_{n-1} \vert x_{n-1}]}$ \\
            derive $\dfrac{P}{F(t_1, \dots, t_n) : \Delta[t_1 \vert x_1, \dots, t_n \vert x_n]}$.
        \item[SI1] If $Q$ is a context, then from $\dfrac{y_1 : \Omega_1, \dots, y_n : \Omega_n}{\Omega = \Omega'}$ and from \\
            $\dfrac{Q}{s_1 = s_2 : \Omega_1}, \dfrac{Q}{s_2 = s'_2 : \Omega_2[s_1 \vert y_1]}, \dots, \dfrac{Q}{s_m = s'_m : \Omega_m[s_1 \vert y_1, \dots, s_{m-1} \vert y_{m-1}]}$ \\
            derive $\dfrac{Q}{\Omega[s_1 \vert y_1, \dots, s_m \vert y_m]} = \Omega'[s'_1 \vert y_1, \dots, s'_m \vert y_m]$.
        \item[SI2] If $Q$ is a context, then from $\dfrac{y_1 : \Omega_1, \dots, y_m : \Omega_m}{s = s' : \Omega}$ and from \\
            $\dfrac{Q}{s_1 = s'_1 : \Omega_1}, \dfrac{Q}{s_2 = s'_2 : \Omega_2[s_1 \vert y_1]}, \dots, \dfrac{Q}{s_m = s'_m : \Omega_m[s_1 \vert y_1, \dots, s_{m-1} \vert y_{m-1}]}$ \\
            derive $\dfrac{Q}{s[s_1 \vert y_1, \dots, s_m \vert y_m] = s'[s'_1 \vert y_1, \dots, s'_m \vert y_m] : \Omega[s_1 \vert y_1, \dots, s_m \vert y_m]}$.
        \item[A1] If $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta = \Delta'}$ is an axiom then from \\
            $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{is a type}}$ and $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta' }$.
        \item[A2] If $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t = t' : \Delta}$ is an axiom, then from \\
            $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t : \Delta}$ and $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t' : \Delta}$ derive $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t = t' : \Delta}$
    \end{itemize}
\end{definition}

\begin{definition}
    A pretheory is \textbf{well-formed} if all of its introductory rules and axioms are well-formed.
    We define a \textbf{Generalized Algebraic Theory} as a well-formed pretheory.
\end{definition}

\section{Models and Homomorphisms}

For now on, we will consider trees to be trees with a unique least element. 
If $\Theta$ is a tree, and $A$ is a node of $\Theta$, then we will write $A \in \Theta$. 
Also, if $B$ is a node of $\Theta$ and $B$ succeeds $A$, then we write \textit{$A \vartriangleleft B$ in $\Theta$}.
The least node of a tree will be denoted as $1$.
For any node $A$ of a tree $\Theta$, such that $A \neq 1$,
there exists a unique $n \geq 0$, and a unique sequence $A_1, \dots, A_n \in \Theta$ such that
$1 \vartriangleleft A_1 \vartriangleleft A_2 \vartriangleleft \dots \vartriangleleft A_n \vartriangleleft A$ in $\Theta$.

The set of contexts of a theory is structured as a tree
where the least element is the empty context and for all $n \geq 1$
the predecessor of $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
is $\langle x_1 : \Delta_1, \dots, x_{n-1} : \Delta_{n-1} \rangle$.

The \textbf{tree of families} has as nodes the collection of sets, families of sets, families of families of sets, \dots
Consider $1 \vartriangleleft A_1 \vartriangleleft A_n \vartriangleleft A$ in the tree of families,
then for all $a_1 \in A_1$, \dots, and for all $a_n \in A_n(a_1, \dots, a_{n-1})$, $A(a_1, \dots, a_n)$ is a set,
and we define an \textbf{operator} at $A$ as a mapping $\lambda a_1 \in A_1. \dots \lambda a_n \in A_n. f(a_1, \dots, a_n)$
such that $\forall a_1 \in A_1, \dots, \forall a_n \in A_n(a_1, \dots, a_{n-1}), f(a_1, \dots, a_n) \in A(a_1, \dots, a_n)$.
We will write $g(a_1, \dots, a_n)$ for the value of the operator $g$ with arguments $a_1, \dots, a_n$.

When we consider a model $M$ of some theory $U$, the derived rules
are interpreted in $M$, not the expressions.
The interpretation of a rule in a model $M$ is written as that rule superscripted by $M$.

Consider $U$ a theory with a model $M$, and $R$ a derived $T$-rule of such theory $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{ is a type}}$
Let $R_i$ be the rule $\dfrac{x_1 : \Delta_1, \dots, x_{i-1} : \Delta_{i-1}}{\Delta_i \text{ is a type}}$ for $1 \leq i \leq n$.
Then $R_1^M$ is a set, $R_2^M$ is a $R_1^M$ indexed family of sets, \dots 
and we have $1 \vartriangleleft R_1^M \vartriangleleft R_2^M \vartriangleleft \dots \vartriangleleft R_n^M \vartriangleleft R^M$.
If instead we take $R_t$, a derived rule of $U$ of the form $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t : \Delta}$, 
we would have that $R_t^M$ would be an operator at $R^M$.

Let $M, M'$ models of $U$, we define a \textbf{model homomorphism} as a mapping $f: M \to M'$
such that 
\begin{enumerate}
    \item for all derived rules $R$ of $U$ of the form $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{is a type}}$,
          there is an operator $f_R$ that for all $a_1 \in R_1^M, \dots, a_n \in R_n^M(a_1, \dots, a_{n-1}), a \in R^M(a_1, \dots, a_n)$,
          $f_R(a_1, \dots, a_n, a) \in R^{M'}(f_{R_1}(a_1), \dots, f_{R_n}(a_1, \dots, a_n))$;
    \item for all derived rules $R_t$ of the form $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t : \Delta}$,
          $\forall a_1 \in R_1^M, \dots, a_n \in R_n^M(a_1, \dots, a_n)$, 
          \begin{equation*}
            f_R(a_1, \dots, R_t^M(a_1, \dots, a_n)) = R_t^{M'}(f_{R_1}(a_1), \dots, f_{R_n}(a_1, \dots, a_n))
          \end{equation*}
    \item for all derived rules of the form $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta = \Delta'}$
          then $f_R = f_{R'}$ 
          where $R$ is $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{is a type}}$
          and $R'$ is $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta' \text{is a type}}$.
\end{enumerate} 

Let $S$ be the set of sort symbols of $U$, to define a homomorphism 
we must ask for a family $\lambda A \in S. f_A$ of operators of correct types,
then for every $T$-rule $R$ of $U$, the operator $f_R$ is defined by induction.

We define the category of models and homomorphisms of a theory $U$ as $U-alg$.

\section{The Category $GAT$}

Let $A_U$ be the alphabet of a theory $U$ and $v_1, v_2, \dots$ a set of variables $V$.

\begin{definition}
    a \textbf{preinterpretation} $I$ of $U$ in $U'$ consists of a function
    \begin{equation*}
        I: A_U \longrightarrow Expressions \ U'
    \end{equation*}
\end{definition}

If $I$ is a preinterpretation of $U$ in $U'$, then we define a function
\begin{equation*}
    \dot{I}: Expressions \ U \longrightarrow Expressions \ U'
\end{equation*}
by induction on the following clauses:
\begin{enumerate}
    \item if $x \in V$, then $\dot{I}(x) = x$;
    \item if $L \in A_U$, then $\dot{I}(L) = I(L)$;
    \item if $L \in A_U$ and $e_1, \dots, e_n$ are expressions in $U$, then
        \begin{equation*}
            \dot{I}(L(e_1, \dots, e_n)) = I(L)[\dot{I}(e_1) | v_1, \dots, \dot{I}(e_n) | v_n]
        \end{equation*}
\end{enumerate}

Then, we can define a function
\begin{equation*}
    \hat{I}: Rules \ U \longrightarrow Rules \ U'
\end{equation*}
by 
\begin{enumerate}
    \item $\hat{I}\Big( \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{ is a type}}\Big) = \dfrac{x_1 : \dot{I}(\Delta_1), \dots, x_n : \dot{I}(\Delta_n)}{\dot{I}(\Delta) \text{ is a type}}$,
    \item $\hat{I}\Big( \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t : \Delta}\Big) = \dfrac{x_1 : \dot{I}(\Delta_1), \dots, x_n : \dot{I}(\Delta_n)}{\dot{I}(t) : \dot{I}(\Delta)}$,
    \item $\hat{I}\Big( \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta = \Delta'}\Big) = \dfrac{x_1 : \dot{I}(\Delta_1), \dots, x_n : \dot{I}(\Delta_n)}{\dot{I}(\Delta) = \dot{I}(\Delta')}$,
    \item $\hat{I}\Big( \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t = t' : \Delta}\Big) = \dfrac{x_1 : \dot{I}(\Delta_1), \dots, x_n : \dot{I}(\Delta_n)}{ \dot{I}(t) = \dot{I}(t') : \dot{I}(\Delta)}$.
\end{enumerate}

\begin{definition}
    An \textbf{interpretation} $I$ of $U$ in $U'$ is a preinterpretation $I$ of $U$ in $U'$
    such that for all introductory rules and axioms $R$ of $U$, $\hat{I}(R)$ is a derived rule of $U'$.
\end{definition}
It can be shown that if $I$ is an interpretation, then for any derived rule $R$ of $U$,
$\hat{I}(R)$ is a derived rule of $U'$.


Now we want to create an equivalence relation on the derived $T$-rules and $\in$-rules of a theory $U$
which we call the \textbf{equivalence relation of identity of denotation}
such that if $R \equiv R'$ by this relation, any model $U$ should interpret
$R$ and $R'$ as identical objects.

Semantically we should not distinguish between the rules
$\dfrac{x_1 : \Delta_1, \dots, x_n: \Delta_n}{\Delta \text{ is a type}}$
and $\dfrac{x_1' : \Delta, \dots, x_n' : \Delta[x_1' | x_1, \dots, x_{n-1}' | x_{n-1}]}{\Delta[x_1' | x_1, \dots, x_n' | x_n] \text{ is a type}}$
as they only differ on the variable names.
Similarly, we should not distinguish between 
$\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta \text{ is a type}}$
and $\dfrac{x_1 : \Delta_1', \dots, x_n : \Delta_n'}{\Delta' \text{ is a type}}$,
if for each $i$, $1 \leq i \leq n$,
$\dfrac{x_1 : \Delta_1, \dots, x_{i-1} : \Delta_{i-1}}{\Delta_i = \Delta_i'}$
and $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta = \Delta'}$
are both derived rules.

We define the relation $\equiv$:
\begin{itemize}
    \item between $T$-rules of a theory $U$ by
        \begin{equation*}
            \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta_{n+1} \text{ is a type}}
            \equiv \dfrac{y_1 : \Omega_1, \dots, y_m : \Omega_m}{\Omega_{m+1} \text{ is a type}}
        \end{equation*}
        if, and only if, $n = m$ and for each $1 \leq i \leq n + 1$
        $\dfrac{x_1 : \Delta_1, \dots, x_{i-1} : \Delta_{i-1}}{\Delta_i = \Omega_i[x_1 | y_1, \dots, x_{i-1} | y_{i-1}]}$
        is a derived rule of $U$.
    \item between $\in$-rules of a theory $U$ by
        \begin{equation*}
            \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t : \Delta}
            \equiv \dfrac{y_1 : \Omega_1, \dots, y_m : \Omega_m}{s : \Omega}
        \end{equation*}
        if, and only if 
        \begin{equation*}
            \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{\Delta_{n+1} \text{ is a type}} 
            \equiv \dfrac{y_1 : \Omega_1, \dots, y_m : \Omega_m}{\Omega_{m+1} \text{ is a type}}
        \end{equation*}
        and $\dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t = s[x_1 | y_1, \dots, x_n | y_n] : \Delta}$
        is a derived rule of $U$.
\end{itemize}

If $I$ and $J$ are both interpretations of a theory $U$ in a theory $U'$
then we say $I \equiv J$ if, and only if, 
for all derived $T$-rules and $\in$-rules $R$ on $U$,
$\hat{I}(R) \equiv \hat{J}(R)$.
Then we define $GAT$ as the category of Generalized Algebraic Theories
and equivalence classes of interpretations.

\section{Contexts and Realizations}

\begin{definition}
    If $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
    and $\langle y_1 : \Omega_1, \dots, y_n : \Omega_m \rangle$
    are contexts of a theory $U$, then a \textbf{realization} 
    of $\langle y_1 : \Omega_1, \dots, y_n : \Omega_m \rangle$
    with regard to $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
    is an $m$-tuple $\langle t_1, \dots, t_m \rangle$ 
    such that for each $j$, $1 \leq j \leq m$, the rule
    \begin{equation*}
        \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t_j : \Omega_j[t_1 | y_1, \dots, t_{j-1} | y_{j - 1}]}
    \end{equation*}
    is a derived rule of $U$.
\end{definition}

Let $U$ be a GAT, then the category $R(U)$ of contexts
and realizations of $U$ we have the morphism
$\langle t_1, \dots, t_m \rangle : \langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle \to \langle y_1 : \Omega_1, \dots, y_m : \Omega_m \rangle$
if, and only if, $\langle t_1, \dots, t_m \rangle$ is a realization of 
$\langle y_1 : \Omega_1, \dots, y_n : \Omega_m \rangle$
with regard to $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$.
The identity of an object $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
is the realization $\langle x_1, \dots, x_n \rangle$.
And composition is defined as 
\begin{equation*}
    \langle s_1, \dots s_w \rangle \circ \langle t_1, \dots, t_m \rangle
     = \langle s_1[t_1 | y_1, \dots, t_m | y_m], \dots, s_w[t_1 | y_1, \dots, t_m | y_m] \rangle
\end{equation*}
when $\langle t_1, \dots, t_m \rangle : \langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle \to \langle y_1 : \Omega_1, \dots, y_m : \Omega_m \rangle$
and $\langle s_1, \dots, s_w \rangle : \langle y_1 : \Omega_1, \dots, y_m : \Omega_m \rangle \to \langle z_1 : \Lambda_1, \dots, z_w : \Lambda_w \rangle$.

The set of objects of $R(U)$ is structured as a tree 
with the empty context $\langle \rangle$ as the least element
and with any context $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
is preceded by $\langle x_1 : \Delta_1, \dots, x_{n-1} : \Delta_{n-1} \rangle$.

For any context $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n, x_{n+1} : \Delta_{n+1}, \dots, x_{n+m} : \Delta_{n+m} \rangle$
of $U$,  $\langle x_1, \dots, x_n \rangle$ is a realization of $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$
with regard to $\langle x_1 : \Delta_1, \dots, x_{n+m} : \Delta_{n+m} \rangle$.
We denote this morphism as $p(\langle x_1 : \Delta_1, \dots, x_{n+m} : \Delta_{n+m} \rangle, \langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle)$.
So for all $A, B \in Ob(R(U))$ such that $A \leq B$, 
$p(B, A)$ is defined and $p(B, A): B \to A$ in $R(U)$.

\begin{lemma}
    Let $A = \langle x_1: \Delta_1, \dots, x_n : \Delta_n \rangle$
    and $A' = \langle y_1 : \Omega_1, \dots, y_m: \Omega_m \rangle$,
    if $B = \langle y_1 : \Omega_1, \dots, y_{m+w} : \Omega_{m+w} \rangle$
    and $f = \langle t_1, \dots, t_m \rangle$ 
    is a realization of $A'$ with regard to $A$, then
    \[
    \begin{tikzcd}
        f^*B \arrow[r, "q(f{,} B)"] \arrow[d, "p(f^*B{,} A)"'] & B \arrow[d, "p(B{,} A')"]\\
        A \arrow[r, "f"'] & A'
    \end{tikzcd}
    \]
    is a pullback in $R(U)$, where
    \begin{multline*}
        f^*B = \langle x_1 : \Delta_1, \dots, x_n : \Delta_n, y_{m+1} : \Omega_{m+1}[t_1 | y_1, \dots, t_m | y_m], \\ 
            \dots, y_{m+w} : \Omega_{m+w}[t_1 | y_1, \dots, t_m | y_m] \rangle
    \end{multline*}
    \begin{equation*}
         q(f, B) = \langle t_1, \dots, t_m, y_{m+1}, \dots, y_{m+w} \rangle
    \end{equation*}
\end{lemma}

This motivates the following definition.

\section{Contextual Categories}

\begin{definition}
    A \textbf{contextual category} consists of
    \begin{enumerate}
        \item a category $C$ wih a terminal object $1$,
        \item a tree structured on the objects of $C$
              such that the terminal object $1$ is the unique
              least element of the tree,
        \item for all $A, B \in Ob C$ such that $A \vartriangleleft B$
              a morphism $p(B): B \to A$, it will be written as $B \openrightarrow A$,
        \item for all $A, A' \in Ob C$, for all $f: A \to A'$ in $C$, 
              for all $B \in Ob C$ such that $A' \vartriangleleft B$,
              an object $f*B$ of $C$ and a morphism $q(f, B): f^*B \to B$ 
              such that the diagram
              \[
                \begin{tikzcd}
                    f^*B \arrow[r, "q(f {,} B)"] \arrow[d, "", open] & B \arrow[d, "", open] \\
                    A \arrow[r, "f"] & A'
                \end{tikzcd}
              \]
              is a pullback in $C$ such that
              \begin{enumerate}
                \item for all $A, B \in Ob C$ such that $A \vartriangleleft B$, 
                      $id_A^*B = B$ and $q(id_A, B) = id_B$,
                \item Whenever
                \[
                    \begin{tikzcd}
                        && B \arrow[d, "", open] \\
                        A \arrow[r, "f"] & A' \arrow[r, "f'"] & A''
                    \end{tikzcd}
                \]
                in $C$, then $(f' \circ f)^*B = f^*(f'^*B)$
                and $q(f' \circ f, B) =  q(f', B) \circ q(f, f'^*B)$.
              \end{enumerate}
    \end{enumerate}
\end{definition}

\begin{remark}
    The category $R(U)$ is structured as a contextual category
    by the tree structure and pullbacks mentioned above.
\end{remark}

The context category $R(U)$ has a homomorphic image $C(U)$ induced by
the equivalence relation of intended equivalence of denotation.
$C(U)$ is the category of equivalence classes of contexts and 
equivalence classes of realizations
\begin{itemize}
    \item $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle \equiv \langle y_1 : \Omega_1, \dots, y_m : \Omega_m \rangle$
          if, and only if, 
          \begin{equation*}
                \dfrac{x_1 : \Delta_1, \dots, x_{n-1} : \Delta_{n-1}}{\Delta_n \text{ is a type}} \equiv \dfrac{y_1 : \Omega_1, \dots, y_{m-1} : \Omega_{m-1}}{\Omega_m \text{ is a type}}
          \end{equation*}
    \item let $\langle t_1, \dots, t_m \rangle$ be a realization of 
          $\langle y_1 : \Omega_1, \dots, y_m : \Omega_m \rangle$ 
          with regard to $\langle x_1 : \Delta_1, \dots, x_n : \Delta_n \rangle$,
          and let $\langle s_1, \dots, s_m \rangle$ be a realization of
          $\langle y'_1 : \Omega'_1, \dots, y'_m : \Omega'_m \rangle$ 
          with regard to $\langle x'_1 : \Delta'_1, \dots, x'_n : \Delta'_n \rangle$,
          then $\langle t_1, \dots, t_m \rangle \equiv \langle s_1, \dots, s_m \rangle$
          if, and only if, for each $j$, $1 \leq j \leq m$,
          \begin{equation*}
            \dfrac{x_1 : \Delta_1, \dots, x_n : \Delta_n}{t_j : \Omega_j[t_1 | y_1, \dots, t_{j-1} | y_{j-1}]}
            \equiv
            \dfrac{x'_1 : \Delta'_1, \dots, x'_n : \Delta'_n}{s_j : \Omega'_j[s_1 | y'_1, \dots, s_{j-1} | y'_{j-1}]}
          \end{equation*}
\end{itemize}

Here $Fam$ takes the same role among contextual categories
as $Set$ with categories. We will give $Fam$ the structure of a context category:

First, we should give it the structure of category. 
To do that we shall introduce its morphisms, the identities and composition.
Given $n, m \geq 0$, 
$1 \vartriangleleft A_1 \vartriangleleft \dots \vartriangleleft A_n$
and $1 \vartriangleleft B_1 \vartriangleleft \dots \vartriangleleft B_m$
in $Fam$, then 
\begin{multline*}
Hom_{Fam}(A_n, B_m) \coloneq \{ \langle f_1, \dots, f_m \rangle | f_1, \dots, f_m
\text{ are operators such that for all } j, \\ 
1 \leq j \leq m, \text{ for } a_1 \in A_1, \dots, a_n \in A_n(a_1, \dots, a_{n-1}),  \\
f_j(a_1, \dots, a_n) \in B_j(f_1(a_1, \dots, a_n), \dots, f_{j-1}(a_1, \dots, a_n)) \}    
\end{multline*}
For $n = 0$, then $Hom_{Fam}(1, B_m) = \{ \langle b_1, \dots, b_m \rangle | b_1 \in B_1, \dots, b_m \in B_m(b_1, \dots, b_{m-1})\}$, 
and $1$ is the terminal object of $Fam$ because
$Hom_{Fam}(A_n, 1) = \{ \langle \rangle \}$.
For the composition in $Fam$ consider 
$1 \vartriangleleft A_1 \vartriangleleft \dots \vartriangleleft A_n$,
$1 \vartriangleleft B_1 \vartriangleleft \dots \vartriangleleft B_m$,
$1 \vartriangleleft C_1 \vartriangleleft \dots \vartriangleleft C_w$,
$\langle f_1, \dots, f_m \rangle : A_n \to B_m$ and 
$\langle g_1, \dots, g_w \rangle : B_m \to C_w$. 
Then we define $\langle g_1, \dots, g_w \langle \circ \langle f_1, \dots, f_m \rangle = \langle h_1, \dots, h_w \rangle$
where for all $k$, $1 \leq k \leq w$, $h_k$ is defined as
\begin{equation*}
    h_k(a_1, \dots, a_n) = g_k(f_1(a_1, \dots, a_n), \dots, f_m(a_1, \dots, a_n))
\end{equation*}

Now we should define the projection morphisms, 
let $1 \vartriangleleft A_1 \vartriangleleft \dots \vartriangleleft A_n \vartriangleleft A$,
then we define $p(A): A \to A_n$ in $Fam$ as
\begin{equation*}
    p(A) = \langle h_1, \dots, h_n \rangle ; h_i(a_1, \dots, a_n, a) = a_i
\end{equation*}

Finally, we should define $f*\cdot$ and $q$ such that they meet the pullback condition. 
Given $1 \vartriangleleft A_1 \vartriangleleft \dots \vartriangleleft A_n$
and $1 \vartriangleleft B_1 \vartriangleleft \dots \vartriangleleft B_m \vartriangleleft B$
and $\langle f_1, \dots, f_m \rangle : A_n \to B_m$ in $Fam$,
then we define $\langle f_1, \dots, f_m \rangle^*B$ is defined as
\begin{multline*}
    \lambda a_1 \in A_1. \lambda a_2 \in A_2 (a_1). \dots \lambda a_n \in A_n(a_1, \dots, a_{n-1}). \\
    B(f_1(a_1, \dots, a_n), \dots, f_m(a_1, \dots, a_n))
\end{multline*} 
and $q(\langle f_1, \dots, f_m \rangle, B) = \langle f_1, \dots, f_m, \gamma \rangle$
where $\gamma$ is the operator defined as $\gamma(a_1, \dots, a_n, b) = b$.

\begin{definition}
    If $C$ and $C'$ are contextual categories, 
    then a functor $F: C \to C'$ is a \textbf{contextual functor} if
    \begin{enumerate}
        \item $F(1_C) = 1_{C'}$ and if $A \vartriangleleft B$ in $C$, then $F(A) \vartriangleleft F(B)$ in $C'$;
        \item for all $A \in Ob C$, $F(p(A)) = p(F(A))$;
        \item for all $f$ and $B$ such that $f^*B$ is defined in $C$,
              $F(f^*B) = F(f)^*F(B)$ and \\ $F(q(f, B)) = q(F(f), F(B))$. 
    \end{enumerate}
\end{definition}

$Con$ is the category of contextual categories and contextual functors.
