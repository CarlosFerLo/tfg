\chapter{Derivation Rules}

We denote "from $S_1, \dots, S_n$ derive $S$", where $S_1, \dots, S_n, S$ are rules or sequents, as follows: 
\begin{mathpar}
    \inferrule{
        S_1 \quad \dots \quad S_n
    }{
        S
    }
\end{mathpar}
Also, we write 
\begin{mathpar}
    \vcenter{\hbox{$ 
        \infer={S_1}{S_2} 
    $}}
    \quad \text{abbreviates} \quad
    \inferrule{S_1}{S_2}
    \text{ and }
    \inferrule{S_2}{S_1}
\end{mathpar}

\section{Derivation Rules of Generalized Algebraic Theories} \label{app:derivation-rules-gat}

In this section we present the derivation rules of Generalized Algebraic Theories as presented in \cite{cartmell1986},
but with a more compact notation.

\begin{mathpar}
    \inferrule*[Left=\RuleLabel{LI1}{li1}]
        {\Gamma \vdash A \type}
        {\Gamma \vdash A = A}

    \and

    \inferrule*[Left=\RuleLabel{LI2}{li2}]
        {\Gamma \vdash t \colon A}
        {\Gamma \vdash t = t \colon A}

    \and

    \inferrule*[Left=\RuleLabel{LI3}{li3}]
        {\Gamma \vdash A_1 = A_2}
        {\Gamma \vdash A_2 = A_1}

    \and

    \inferrule*[Left=\RuleLabel{LI4}{li4}]
        {\Gamma \vdash t_1 = t_2 \colon A}
        {\Gamma \vdash t_2 = t_1 \colon A}

    \and

    \inferrule*[Left=\RuleLabel{LI5}{li5}]
        {\Gamma \vdash A_1 = A_2 \\ \Gamma \vdash A_2 = A_3}
        {A_1 = A_3}

    \and

    \inferrule*[Left=\RuleLabel{LI6}{li6}]
        {\Gamma \vdash t_1 = t_2 \colon A \\ \Gamma \vdash t_2 = t_3 \colon A}
        {\Gamma \vdash t_1 = t_3 \colon A}

    \and 

    \inferrule*[Left=\RuleLabel{LI7}{li7}]
        {\Gamma \vdash t_1 = t_2 \colon A_1 \\ \Gamma \vdash A_1 = A_2}
        {t_1 = t_2 \colon A_2}

    \and 

    \inferrule*[Left=\RuleLabel{T1}{t1}]
        {\Gamma \vdash A_1 = A_2 \quad \Gamma \vdash t \colon A_1}
        {\Gamma \vdash t \colon A_2}
\end{mathpar}
\begin{enumerate}
    \ruleitem{CF1}{cf1} For $n \geq 0, 1 \leq i \leq n+1$: \begin{mathpar}
            \inferrule{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash A_{n+1} \type
            }{
                x_1 \colon A_1, \dots, x_{n+1} \colon A_{n+1} \vdash x_i \colon A_i   
            }
        \end{mathpar}
        providing that $x_{n+1}$ is a variable distinct from all $x_1, \dots, x_n$.
    \ruleitem{CF2(a)}{cf2a} For every sort symbol $S$ with well-formed introductory rule \[
            x_1 \colon A_1, \dots, x_n \colon A_n 
            \vdash
            S(x_1, \dots, x_n) \type,
        \] for every context $\Gamma$, \begin{mathpar}
            \inferrule{
                \Gamma \vdash t_1 \colon A_1 \\
                \Gamma \vdash t_2 \colon A_2[x_1  \leftarrow  t_1] \\
                \dots \\
                \Gamma \vdash t_n \colon A_n[x_1  \leftarrow  t_1, \dots, x_{n-1}  \leftarrow  t_{n-1}]
            }{
                \Gamma \vdash S(t_1, \dots, t_n) \type
            }
        \end{mathpar}
    \ruleitem{CF2(b)}{cf2b} For every operator symbol $F$ with well-formed introductory rule \[
            x_1 \colon A_1, \dots, x_n \colon A_n 
            \vdash
            F(x_1, \dots, x_n) \colon A,
        \] for every context $P$, \begin{mathpar}
            \inferrule{
                \Gamma \vdash t_1 \colon A_1 \\
                \Gamma \vdash t_2 \colon A_2[x_1  \leftarrow  t_1] \\
                \dots \\
                \Gamma \vdash t_n \colon A_n[x_1  \leftarrow  t_1, \dots, x_{n-1}  \leftarrow  t_{n-1}]
            }{
                \Gamma \vdash F(t_1, \dots, t_n) \colon A[x_1  \leftarrow  t_1, \dots, x_n  \leftarrow  t_n]
            }
        \end{mathpar}
    \ruleitem{SI1}{si1} If $\Gamma$ is a context, then \begin{mathpar}
            \inferrule{
                x_1 \colon A_1, \dots x_n \colon A_n \vdash A = A' \\
                \Gamma \vdash s_1 = s'_1 \colon A_1 \\
                \Gamma \vdash s_2 = s'_2 \colon A_2[x_1  \leftarrow  s_1] \\
                \dots \\
                \Gamma \vdash s_m = s'_m \colon A_n[x_1  \leftarrow  s_1, \dots, x_{n-1}  \leftarrow  s_{n-1}]
            }{
                \Gamma 
                \vdash
                A[x_1  \leftarrow  s_1, \dots, x_n  \leftarrow  s_n]
                =
                A'[x_1  \leftarrow  s'_1, \dots, x_n  \leftarrow  s'_n]
            }
        \end{mathpar}
    \ruleitem{SI2}{si2} If $\Gamma$ is a context, then \begin{mathpar}
            \inferrule{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash s = s' \colon A \\
                \Gamma \vdash s_1 = s'_1 \colon A_1 \\
                \Gamma \vdash s_2 = s'_2 \colon A_2 [x_1  \leftarrow  s_1] \\
                \dots \\
                \Gamma \vdash s_n = s'_n \colon A_n[x_1  \leftarrow  s_1, \dots, x_{n-1}  \leftarrow  s_{n-1}]
            }{
                s[x_1  \leftarrow  s_1, \dots, x_n  \leftarrow  s_n]
                = 
                s'[x_1  \leftarrow  s'_1, \dots, x_n  \leftarrow  s'_n]
            }
        \end{mathpar}
    \ruleitem{A1}{a1} If $x_1 \colon A_1, \dots, x_n \colon A_n \vdash A = A'$ is an axiom, then \begin{mathpar}
            \inferrule{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash A \type \\
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash A' \type   
            }{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash A = A'
            }
        \end{mathpar}
    \ruleitem{A2}{a2} If $x_1 \colon A_1, \dots, x_n \colon A_n \vdash t = t' \colon A$ is an axiom, then from \begin{mathpar}
            \inferrule{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash t \colon A  \\
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash t' \colon A
            }{
                x_1 \colon A_1, \dots, x_n \colon A_n \vdash t = t' \colon A
            }
        \end{mathpar}
\end{enumerate}

\section{Derivation Rules of Calculus of Constructions} \label{app:derivation-rules-coc}

In this section we present all the derivation rules for CoC as presented in \cite{streicher}.

Context Formation Rules:
\begin{mathpar}
    \RuleLabel{Empty}{empty} \quad
    \inferrule
        { }
        {\quad \ok}

    \and

    
    \RuleLabel{Cont-Intro}{cont-intro} \quad
    \vcenter{\hbox{$
        \infer=
        {\Gamma \vdash A \type}
        {\Gamma, x \colon A \ok} 
    $}}
    \text{, if} x \notin \Var(\Gamma)

    \and
    
    \RuleLabel{Cont-Thin}{cont-thin} \quad
    \inferrule
        {\Gamma, \Delta \ok \\ \Gamma \vdash A \type}
        {\Gamma, x \colon A, \Delta \ok}
    \text{, if} x \notin \Var(\Gamma) \cup \Var(\Delta)
    
    \and

    \inferrule*[Left=\RuleLabel{Cont-Sub}{cont-sub}]
        {\Gamma, x \colon A, \Delta \ok \\ \Gamma \vdash t \colon A}
        {\Gamma, \Delta[x  \leftarrow  t] \ok}
\end{mathpar}

Context Equality Rules:
\begin{mathpar}
    \RuleLabel{Crefl}{crefl} \quad
    \vcenter{\hbox{$
        \infer=
            {\Gamma \ok}
            {\Gamma = \Gamma}
    $}}

    \and

    \RuleLabel{Csym}{csym} \quad 
        \inferrule
            {\Gamma = \Delta}
            {\Delta = \Gamma}

    \and

    \RuleLabel{Ctrans}{ctrans} \quad
        \inferrule
            {\Gamma = \Delta \\ \Delta = \Theta}
            {\Gamma = \Theta}

    \and

    \RuleLabel{Cequ}{cequ} \quad
        \inferrule
            {\Gamma, x \colon A, \Delta \ok \\ \Gamma \vdash A = B}
            {\Gamma, x \colon A, \Delta = \Gamma, x \colon B, \Delta}
    
    \and

    \RuleLabel{Creflect1}{creflect1} \quad
        \inferrule
            {\Gamma, x \colon A = \Delta, x \colon B}
            {\Gamma = \Delta}

    \and

    \RuleLabel{Creflect2}{creflect2} \quad
        \inferrule
            {\Gamma, x \colon A = \Delta, x \colon B}
            {\Gamma \vdash A = B}

\end{mathpar}

Structural Rules for Judgement Formation
\begin{mathpar}
    \RuleLabel{Ass}{ass} \quad
        \inferrule
            {\Gamma, x \colon A, \Delta \ok}
            {\Gamma, x \colon A, \Delta \vdash x \colon A}
    
    \and

    \RuleLabel{Reflection1}{reflection1} \quad
        \inferrule
            {\Gamma \vdash \mathcal{J}}
            {\Gamma \ok}

    \and

    \RuleLabel{Reflection2}{reflection2} \quad
        \inferrule
            {\Gamma \vdash t : A}
            {\Gamma \vdash A \type}

    \and

    \RuleLabel{Thin}{thin} \quad
        \inferrule
            {\Gamma, \Delta \vdash \mathcal{J} \\ \Gamma \vdash A \type}
            {\Gamma, x \colon A, \Delta \vdash \mathcal{J}}
        \text{, if } x \notin \Var(\Gamma) \cup \Var(\Delta)

    \and

    \RuleLabel{Sub}{sub} \quad
        \inferrule
            {\Gamma, x \colon A, \Delta \vdash \mathcal{J} \\ \Gamma \vdash t : A}
            {\Gamma, \Delta[x  \leftarrow  t] \vdash \mathcal{J}[x  \leftarrow  t]}
\end{mathpar}

Equality Rules:
\begin{mathpar}
    \RuleLabel{Equ}{equ} \quad
        \inferrule 
            {\Gamma = \Delta \\ \Gamma \vdash \mathcal{J}}
            {\Delta \vdash \mathcal{J}}

    \and

    \RuleLabel{Refl-Typ}{refl-typ} \quad
        \vcenter{\hbox{$
            \infer=
                {\Gamma \vdash A = A}
                {\Gamma \vdash A \type}
        $}}

    \and

    \RuleLabel{Refl-Obj}{refl-obj} \quad 
        \vcenter{\hbox{$
            \infer=
                {\Gamma \vdash t = t : A}
                {\Gamma \vdash t : A}
        $}}

    \and

    \RuleLabel{Symm-Typ}{symm-typ} \quad
        \inferrule 
            {\Gamma \vdash A = B}
            {\Gamma \vdash B = A}

    \and

    \RuleLabel{Symm-Obj}{symm-obj} \quad
        \inferrule
            {\Gamma \vdash t = s \colon A}
            {\Gamma \vdash s = t \colon A}

    \and

    \RuleLabel{Trans-Typ}{trans-typ} \quad
        \inferrule
            {\Gamma \vdash A = B \\ \Gamma \vdash B = C}
            {\Gamma \vdash A = C}

    \and

    \RuleLabel{Trans-Obj}{trans-obj} \quad
        \inferrule 
            {\Gamma \vdash t_1 = t_2 \colon A \\ \Gamma \vdash t_2 = t_3 \colon A}
            {\Gamma \vdash t_1 = t_3 \colon A} 

    \and

    \RuleLabel{Repl1}{repl1} 
        \inferrule
            {\Gamma \vdash t = s \colon A \\ \Gamma, x \colon A, \Delta \ok}
            {\Gamma, \Delta[x  \leftarrow  t] = \Gamma, \Delta[x  \leftarrow  t]}
\end{mathpar}

\begin{enumerate}
    \ruleitem{Repl2}{repl2} $\dfrac{
        \Gamma \vdash t = s : A
        \quad
        \Gamma, x \colon A, \Delta \vdash B \type
    }{
        \Gamma, \Delta[x  \leftarrow  t]
        \vdash
        B[x  \leftarrow  t] = B[x  \leftarrow  s]
    }$
    \ruleitem{Repl3}{repl3} $\dfrac{
        \Gamma \vdash t_1 = t_2 : A
        \quad
        \Gamma, x \colon A, \Delta \vdash s : B
    }{
        \Gamma, \Delta[ x  \leftarrow  t_1]
        \vdash
        s[x  \leftarrow  t_1] = s[x  \leftarrow  t_2] : B[x  \leftarrow  t_1]
    }$
    \ruleitem{Conv1}{conv1} $\dfrac{
        \Gamma \vdash A = B
        \quad
        \Gamma \vdash t : A
    }{
        \Gamma \vdash t : B
    }$
    \ruleitem{Conv2}{conv2} $\dfrac{
        \Gamma \vdash A = B
        \quad
        \Gamma \vdash t = s : A
    }{
        \Gamma \vdash t = s : B
    }$
\end{enumerate}

Rules for Products of types

\begin{enumerate}
    \ruleitem{$\Pi$-Form}{prod-form} $\dfrac{
        \Gamma, x \colon A \vdash B \type
    }{
        \Gamma 
        \vdash 
        (\Pi x \colon A) B \type
    }$
    \ruleitem{$\Pi$-Equ}{prod-equ} $\dfrac{
        \Gamma \vdash A_1 = A_2
        \quad
        \Gamma, x \colon A_1 \vdash B_1 = B_2
    }{
        \Gamma
        \vdash
        (\Pi x \colon A_1) B_1
        =
        (\Pi x \colon A_2) B_2
    }$
    \ruleitem{$\Pi$-Intro}{prod-intro} $\dfrac{
        \Gamma, x \colon A \vdash t : B
    }{
        \Gamma
        \vdash
        (\lambda x \colon A) t
        :
        (\Pi x \colon A) B
    }$
    \ruleitem{$\xi$-Rule}{xi-rule} $\dfrac{
        \Gamma, x \colon A_1 \vdash t_1 = t_2 : B
        \quad
        \Gamma \vdash A_1 = A_2
    }{
        \Gamma
        \vdash
        (\lambda x \colon A_1) t_1
        =
        (\lambda x \colon A_2) t_2
        \colon
        (\Pi x \colon A_1) B
    }$
    \ruleitem{$\Pi$-Elim}{prod-elim} $\dfrac{
        \Gamma \vdash t \colon (\Pi x \colon A) B
        \quad
        \Gamma \vdash s \colon A
        \quad
        \Gamma, x \colon A \vdash B \type
    }{
        \Gamma
        \vdash
        \App([x \colon A] B, t, s) 
        \colon 
        B[x  \leftarrow  s]
    }$
    \ruleitem{$\Pi$-Elim-Equ}{prod-elim-equ} $\dfrac{
        \Gamma \vdash A_1 = A_2
        \quad
        \Gamma \vdash t_1 = t_2 \colon (\Pi x \colon A_1) B_1
        \quad
        \Gamma, x \colon A_1 \vdash B_1 = B_2
        \quad
        \Gamma \vdash s_1 = s_2 \colon A_1
    }{
        \Gamma
        \vdash
        \App([x \colon A_1] B_1, t_1, s_1)
        = 
        \App([x \colon A_2] B_2, t_2, s_2)
        \colon
        B_1[x  \leftarrow  s_1]
    }$
    \ruleitem{$\beta$-Rule}{beta-rule} $\dfrac{
        \Gamma, x \colon A \vdash B \type
        \quad
        \Gamma, x \colon A \vdash t \colon B
        \quad
        \Gamma \vdash s \colon A
    }{
        \Gamma 
        \vdash
        \App([x \colon A] B, (\lambda x : A) t, s)
        = 
        t[x  \leftarrow  s] 
        \colon
        B[x  \leftarrow  s]
    }$
    \ruleitem{$\eta$-rule}{eta-rule} $\dfrac{
        \Gamma, x \colon A \vdash B \type
        \quad
        \Gamma \vdash t \colon (\Pi x \colon A) B
    }{
        \Gamma 
        \vdash
        (\lambda x \colon A) \App([x \colon A] B, t, x) 
        = 
        t
        \colon
        (\Pi x \colon A) B
    }$
\end{enumerate}

Rules for Propositions

\begin{enumerate}
    \ruleitem{Prop}{prop} $\dfrac{
        \Gamma \ok
    }{
        \Gamma \vdash \Prop \type
    }$ 
    \ruleitem{Prop-Type}{prop-type} $\ddoublefrac{
        \Gamma \vdash p \colon \Prop
    }{
        \Gamma \vdash \Proof(p) \type
    }$
    \ruleitem{Prop-Equ}{prop-equ} $\ddoublefrac{
        \Gamma \vdash p_1 = p_2 \colon \Prop
    }{
        \Gamma \vdash \Proof(p_1) = \Proof(p_2)
    }$
    \ruleitem{$\forall$-Intro}{forall-intro} $\dfrac{
        \Gamma, x \colon A 
        \vdash
        p \colon \Prop
    }{
        \Gamma
        \vdash
        (\forall x \colon A) p 
        \colon \Prop
    }$
    \ruleitem{$\forall$-Equ}{forall-equ} $\dfrac{
        \Gamma, x \colon A_1 \vdash p_1 = p_2 \colon \Prop
        \quad
        \Gamma \vdash A_1 = A_2
    }{
        \Gamma
        \vdash
        (\forall x \colon A_1) p_1
        =
        (\forall x \colon A_2) p_2 
        \colon
        \Prop
    }$
    \ruleitem{$\forall$-Elim}{forall-elim} $\dfrac{
        \Gamma, x \colon A \vdash p \in \Prop
    }{
        \Gamma 
        \vdash
        \Proof((\forall x \colon A) p)
        = 
        (\Pi x \colon A) \Proof(p)
    }$
\end{enumerate}
